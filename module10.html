N = 100; // total number of data points x = new Array(N); // x protein concentration (uM) y = new Array(N); // y protein concentration (uM) pos_y = new Array(N); // positively autoregulated // y protein concentration (uM) neg_y = new Array(N); // neagtively autoregulated // y protein concentration (uM) X = 10; // constant x protein concentration x[0] = X; // initial x protein concentration y[0] = 0; // initial y protein concentration pXY = 0.2; // production parameter pY = 0.9; // degradation/dilution parameter par = 0.5; // positive autoregulation factor nar = 0.5; // negative autoregulation facotr for (n = 1; n < N; n++) // N-1 loops (1 loop = 1 min) { x[n] = X; // constant x protein concentration y[n] = pXY*x[n-1] + pY*y[n-1]; // update y protein concentration // update positvely autoregulated y protein concentration pos_y[n]= pXY*x[n-1] + (pY + par)*y[n-1]; // update negatively autoregulated protein concentration neg_y[n]= pXY*x[n-1] + (pY - nar)*y[n-1]; } // end of for loop
